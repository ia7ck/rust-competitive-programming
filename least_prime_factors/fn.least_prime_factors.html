pub fn least_prime_factors(n: usize) -> Vec<usize>

「k を割る最小の素数」をエラトステネスのふるいの要領で 2 以上 n 未満の全ての k について計算します。参考

Examples

use least_prime_factors::least_prime_factors;
let facs = least_prime_factors(10);
assert_eq!(facs[2], 2);
assert_eq!(facs[3], 3);
assert_eq!(facs[4], 2);
assert_eq!(facs[5], 5);
assert_eq!(facs[6], 2);
assert_eq!(facs[7], 7);
assert_eq!(facs[8], 2);
assert_eq!(facs[9], 3);
