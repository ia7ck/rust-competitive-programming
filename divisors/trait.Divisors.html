fn divisors(self) -> Vec<Self>

非負整数の約数を昇順で返します。0 に対しては空のベクタ vec![] を返します。

§Examples
use divisors::Divisors;

assert_eq!(24_u32.divisors(), vec![1, 2, 3, 4, 6, 8, 12, 24]);
